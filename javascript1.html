Loops in JavaScript

for Loop

The for loop repeats until a specified condition evaluates to false.

// Basic for loop
for (let i = 0; i < 5; i++) {
  console.log("Iteration: " + i);
}

// Looping through an array
const fruits = ["apple", "banana", "orange"];
for (let i = 0; i < fruits.length; i++) {
  console.log(fruits[i]);
}
while Loop

The while loop executes as long as the condition is true.

let count = 0;
while (count < 3) {
  console.log("Count: " + count);
  count++;
}

// Do-while loop (executes at least once)
let num = 5;
do {
  console.log("Number: " + num);
  num--;
} while (num > 0);
for...of Loop

The for...of loop creates a loop iterating over iterable objects like arrays.

const colors = ["red", "green", "blue"];

for (const color of colors) {
  console.log(color);
}

// With index using entries()
for (const [index, color] of colors.entries()) {
  console.log(index + ": " + color);
}

Array Methods

Common Array Methods

Arrays have many built-in methods for manipulation and iteration.

const numbers = [1, 2, 3, 4, 5];

// map - creates new array with transformed elements
const doubled = numbers.map(num => num * 2);
console.log(doubled); // [2, 4, 6, 8, 10]

// filter - creates new array with elements that pass test
const even = numbers.filter(num => num % 2 === 0);
console.log(even); // [2, 4]

// reduce - reduces array to single value
const sum = numbers.reduce((acc, num) => acc + num, 0);
console.log(sum); // 15

// find - returns first element that passes test
const firstEven = numbers.find(num => num % 2 === 0);
console.log(firstEven); // 2
More Array Methods
const fruits = ["apple", "banana", "orange", "grape"];

// forEach - executes function for each element
fruits.forEach(fruit => console.log(fruit));

// some - tests if at least one element passes test
const hasApple = fruits.some(fruit => fruit === "apple");
console.log(hasApple); // true

// every - tests if all elements pass test
const allStrings = fruits.every(fruit => typeof fruit === "string");
console.log(allStrings); // true

// includes - checks if array contains element
console.log(fruits.includes("banana")); // true

Working with Objects

Object Basics

Objects are collections of key-value pairs.

// Object creation
const person = {
  firstName: "John",
  lastName: "Doe",
  age: 30,
  hobbies: ["reading", "coding", "gaming"],
  address: {
    street: "123 Main St",
    city: "Boston"
  },
  getFullName: function() {
    return this.firstName + " " + this.lastName;
  }
};

// Accessing properties
console.log(person.firstName); // John
console.log(person["lastName"]); // Doe
console.log(person.getFullName()); // John Doe
console.log(person.hobbies[0]); // reading
Object Methods
const car = {
  brand: "Toyota",
  model: "Camry",
  year: 2020
};

// Object.keys - get all keys
console.log(Object.keys(car)); // ["brand", "model", "year"]

// Object.values - get all values
console.log(Object.values(car)); // ["Toyota", "Camry", 2020]

// Object.entries - get key-value pairs
console.log(Object.entries(car));
// [["brand", "Toyota"], ["model", "Camry"], ["year", 2020]]

// Adding properties
car.color = "blue";
console.log(car.color); // blue

DOM Manipulation

Selecting Elements

The DOM (Document Object Model) allows JavaScript to interact with HTML.

// Selecting elements
const header = document.getElementById("header");
const buttons = document.getElementsByClassName("btn");
const firstPara = document.querySelector("p");
const allDivs = document.querySelectorAll("div");

// Creating elements
const newDiv = document.createElement("div");
newDiv.textContent = "Hello, World!";
newDiv.classList.add("my-class");

// Appending to DOM
document.body.appendChild(newDiv);
Event Handling
// Adding event listeners
const button = document.getElementById("myButton");

button.addEventListener("click", function() {
  console.log("Button clicked!");
  this.style.backgroundColor = "red";
});

// Event delegation
document.addEventListener("click", function(event) {
  if (event.target.classList.contains("item")) {
    console.log("Item clicked: " + event.target.textContent);
  }
});

ES6+ Features

Destructuring

Destructuring allows unpacking values from arrays or properties from objects.

// Array destructuring
const colors = ["red", "green", "blue"];
const [firstColor, secondColor] = colors;
console.log(firstColor); // red
console.log(secondColor); // green

// Object destructuring
const person = {
  name: "Alice",
  age: 25,
  city: "New York"
};
const { name, age } = person;
console.log(name); // Alice
console.log(age); // 25
Spread Operator & Template Literals
// Spread operator
const arr1 = [1, 2, 3];
const arr2 = [4, 5, 6];
const combined = [...arr1, ...arr2];
console.log(combined); // [1, 2, 3, 4, 5, 6]

// With objects
const obj1 = { a: 1, b: 2 };
const obj2 = { c: 3, d: 4 };
const merged = { ...obj1, ...obj2 };
console.log(merged); // {a: 1, b: 2, c: 3, d: 4}

// Template literals
const userName = "John";
const greeting = `Hello, ${userName}! Welcome to our app.`;
console.log(greeting); // Hello, John! Welcome to our app.
